const [authState, setAuthState] = useState<AuthState>('login');
const [user, setUser] = useState<UserProfile | null>(null);
const [isLoading, setIsLoading] = useState(false);
const [hydrated, setHydrated] = useState(false);

useEffect (empty dep array — runs once on mount, AUTH-03):
```typescript
useEffect(() => {
  try {
    const savedUser = localStorage.getItem('jumpin_user');
    if (savedUser) {
      setUser(JSON.parse(savedUser));
      setAuthState('dashboard');
    }
  } catch {
    localStorage.removeItem('jumpin_user'); // Clear corrupted data
  } finally {
    setHydrated(true);
  }
}, []);
```

Hydration guard — return null before hydration completes (prevents login-form flash on refresh):
```typescript
if (!hydrated) return null;
```

handleLogin (AUTH-01):
```typescript
const handleLogin = (email: string, password: string, onError: (msg: string) => void) => {
  setIsLoading(true);
  setTimeout(() => {
    if (email === 'demo@example.com' && password === 'password123') {
      const mockUser: UserProfile = {
        id: 'demo-123',
        first_name: 'Demo',
        last_name: 'User',
        email: 'demo@example.com',
        school: 'JumpIn Testing School',
        dob: '2000-01-01',
        last_checkin: undefined,
      };
      setUser(mockUser);
      localStorage.setItem('jumpin_user', JSON.stringify(mockUser));
      setAuthState('dashboard');
    } else {
      onError('Credenziali non valide. Usa demo@example.com / password123');
    }
    setIsLoading(false);
  }, 1200);
};
```

handleRegister (AUTH-02):
```typescript
const handleRegister = (profile: Omit<UserProfile, 'id' | 'last_checkin'>) => {
  setIsLoading(true);
  setTimeout(() => {
    const mockUser: UserProfile = {
      id: Math.random().toString(36).substr(2, 9),
      ...profile,
      last_checkin: undefined,
    };
    setUser(mockUser);
    localStorage.setItem('jumpin_user', JSON.stringify(mockUser));
    setAuthState('dashboard');
    setIsLoading(false);
  }, 2000);
};
```

handleLogout (AUTH-04):
```typescript
const handleLogout = () => {
  setUser(null);
  localStorage.removeItem('jumpin_user');
  setAuthState('login');
};
```

handleCheckIn (AUTH-05):
```typescript
const handleCheckIn = () => {
  if (!user) return;
  const updatedUser = { ...user, last_checkin: new Date().toISOString() };
  setUser(updatedUser);
  localStorage.setItem('jumpin_user', JSON.stringify(updatedUser));
};
```

Render logic (after the hydration guard):
```typescript
if (authState === 'login') {
  return (
    <LoginForm
      onLogin={handleLogin}
      isLoading={isLoading}
      onNavigateRegister={() => setAuthState('register')}
    />
  );
}

if (authState === 'register') {
  return (
    <RegisterForm
      onRegister={handleRegister}
      isLoading={isLoading}
      onNavigateLogin={() => setAuthState('login')}
    />
  );
}

if (authState === 'dashboard' && user) {
  return (
    <Dashboard
      user={user}
      onLogout={handleLogout}
      onCheckIn={handleCheckIn}
    />
  );
}

return null;
```
